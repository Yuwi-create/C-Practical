#include <iostream>
using namespace std;

class Calculator {
public:
    
    int add(int a, int b) {
        return a + b;
    }

    
    int add(int a, int b, int c) {
        return a + b + c;
    }
};

int main() {
    
    Calculator calc;

    int sum1 = calc.add(5, 10);  
    cout << "Sum of two integers: " << sum1 << endl;

  
    int sum2 = calc.add(5, 10, 15);  
    cout << "Sum of three integers: " << sum2 << endl;

    return 0;
}
